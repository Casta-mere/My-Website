### 参数

  <TerminalLine>
    <Cmd text="awk" />

### 管道

<TerminalRoot title={"管道"}>
  <TerminalLine>
    <Cmd text="docker ps" />
    <Args text="-a" />
    <Tunnel text="|" />
    <Cmd text="grep" />
    <Line text="neutron" />
  </TerminalLine>
  <TerminalLine>
    <Cmd text="cat" />
    <Line text="/etc/ssh/sshd_config" />
    <Tunnel text="|" />
    <Cmd text="grep" />
    <Line text="-v '^#'" />
    <Tunnel text="|" />
    <Cmd text="grep" />
    <Line text="-v '^$'" />
  </TerminalLine>
  <TerminalLine>
    <Cmd text="cat" />
    <Line text="./id_rsa.pub" />
    <Tunnel text="&gt;&gt;" />
    <Line text="authorized_keys" />
  </TerminalLine>
</TerminalRoot>
